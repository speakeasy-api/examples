# CoffeeOrder

Represents a coffee order in the system

## Example Usage

```typescript
import { CoffeeOrder } from "speakeasy-coffee-client/models/components";

let value: CoffeeOrder = {
  id: 6,
  customerName: "Grace",
  coffeeType: "Cold Brew",
  size: "Medium",
  extras: [
    "Vanilla syrup",
  ],
  price: 5.5,
};
```
